1014  Stream& write_to_stream(Stream& os, View const& sv) {
1015  typename Stream::sentry sentry(os);
1016 
1017  if (!os)
1018  return os;
1019 
1020  const std::streamsize length = static_cast<std::streamsize>(sv.length());
1021 
1022  // Whether, and how, to pad:
1023  const bool pad = (length < os.width());
1024  const bool left_pad = pad && (os.flags() & std::ios_base::adjustfield) == std::ios_base::right;
1025 
1026  if (left_pad)
1027  write_padding(os, os.width() - length);
1028 
1029  // Write span characters:
1030  os.rdbuf()->sputn(sv.begin(), length);
1031 
1032  if (pad && !left_pad)
1033  write_padding(os, os.width() - length);
1034 
1035  // Reset output stream width:
1036  os.width(0);
1037 
1038  return os;
1039  }
